<!DOCTYPE html>
<!-- Document Type Declaration - tells the browser this is an HTML5 document -->

<html lang="en">
<!-- Root element - contains all content on the page. 'lang' attribute specifies the language -->

<head>
	<!-- Head section - contains metadata about the document (not displayed on the page) -->
	
	<meta charset="UTF-8">
	<!-- Character encoding - tells browser how to interpret text characters -->
	
	<meta name="viewport" content="width=device-width, initial-scale=1.0">
	<!-- Viewport meta tag - makes the page responsive on mobile devices -->
	
	<title>My First HTML5 Page</title>
	<!-- Title element - appears in browser tab and search engine results -->
	
</head>

<body>
	<!-- Body section - contains all visible content that appears on the webpage -->
	
	<h1>Welcome to My Website</h1>
	<!-- Main heading - largest heading element for page title -->
	
	<p>This is a paragraph of text in my minimal HTML5 page.</p>
	<!-- Paragraph element - contains a block of text -->
	
	<!-- You can add more content here -->
	
</body>
